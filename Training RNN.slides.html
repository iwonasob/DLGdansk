def build_rnn(n_frames=n_frames, n_features=n_features,
                     n_classes=10):

    # INPUT
    x = Input(shape=(n_frames, n_features), dtype='float32')
  
    y= LSTM(120, return_sequences = True)(x)
    y= Dropout(0.2)(y)  
    
    y= LSTM(120)(y)
    y= Dropout(0.2)(y) 

    y = Dense(n_classes, activation='sigmoid')(y)

    m = Model(inputs=x, outputs=y)
    return m
